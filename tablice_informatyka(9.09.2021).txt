#include <iostream>
using namespace std;

int main()
{
    int tab[12] = { 4, 5, 20, 125, 89, 65, 77, 2, 1, 0, 123, 47 };

    int a, b;

    for (a = 0; a <= 10; a++) {
        for (b = a + 1; b <= 11; b++)
        {
            int numb;

            if (tab[a] > tab[b])
            {
                numb = tab[a];
                tab[a] = tab[b];
                tab[b] = numb;
            }
        }
    }

    for (a = 0; a <= 11; a++) {
        cout << endl << tab[a] << endl;
    }
}
